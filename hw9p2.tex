	\subsection*{Part 2: Tough Decisions}
	For each of the following problems, state (and briefly justify by describing the necessary reduction/TM/both) whether it is decidable (i.e. yes instances are a recursive language), semi-decidable (i.e. yes instances are an RE language) but not decidable, or neither.
	
	\linesep
	
	For each of the following $ n $, let $ L_n $ be the language that accepts encodings $ M $ of Turing machines that satisfy the property $ n $, and $ M^* $ be the Turing machine where $ L(M^*)=L_n $. Let $ K $ be the Turing machine that accepts or rejects $ N\# x $ given $ M^*$ and the reduction $ N' $.
	
	\begin{enumerate}[1.]
	\item \textit{Given a DFA $ D $ and a string $ x $, does $ D $ accept $ x $?}
	
	This is decidable; we can simply run $ D $ on $ x $ and see if it enters an accept state or not.\\
	
	\item Given a DFA $ D $, is $ L(D) $ finite?
	
	This is decidable; we can check if $ D $ loops on any state (if there exists a state $ q $ and a character $ c $ and a transition $ \delta(q,c)=q $), and if this transition is a part of a series of transitions that would accept some string $ x $. (If so, we can pump this character $ c $ into the string as many times as we want to in the right place). Otherwise, the set of strings accepted by $ D $ is fixed. \\
	
	\item \textit{Given a TM $ M $, does $ L(M) $ contain at least 100 elements? }
	
	Not recursively enumerable - we can formulate the reduction $ \bar{MP} \leq L_3$ as follows:
	Given $ N\#x $, run $ N $ on $ x $. Create a TM $ N' $ with input $ y $ with the properties dependent on whether or not $ N $ accepts $ x $:\\
	
	\begin{algorithmic}
		\IF{$ N $ accepts $ x $}
		\RETURN{$ N'$ accepts $ y $}  ($ \implies L(N')=\Sigma^* \implies \abs{L(N')}>100$)
		\ELSE
		\RETURN {$ N' $ rejects $ y $} ($ \implies L(N')=\varnothing \implies \abs{L(N')}=0<100$)
		\ENDIF
	\end{algorithmic}
	Finally, let $ K $ accept $ N\#x $ if and only if $ M^* $ accepts $ N' $.
	\bigskip
	
	\item \textit{Given a TM $ M $, does $ L(M) $ contain at most 100 elements?}
	
	Not recursively enumerable - we can formulate the reduction $ \bar{MP} \leq L_4$ as follows:
	Given $ N\#x $, run $ N $ on $ x $. Create a TM $ N' $ with input $ y $ with the properties dependent on whether or not $ N $ accepts $ x $:\\
	
	\begin{algorithmic}
		\IF{$ N $ accepts $ x $}
		\RETURN {$ N' $ rejects $ y $} ($ \implies L(N')=\varnothing \implies \abs{L(N')}=0<100$)
		\ELSE
		\RETURN{$ N'$ accepts $ y $}  ($ \implies L(N')=\Sigma^* \implies \abs{L(N')}>100$)
		\ENDIF
	\end{algorithmic}
	Finally, let $ K $ accept $ N\#x $ if and only if $ M^* $ accepts $ N' $.\\
	
	\item \textit{Given a TM $ M $, is there any string on which $ M $ halts?}
	
	Not recursively enumerable - we can formulate the reduction $ \bar{HP} \leq L_5$. Given $ N\#x $, run $ N $ on $ x $. Create a TM $ N' $ with input $ y $ with the properties dependent on whether or not $ N $ halts on $ x $:\\
	
	\begin{algorithmic}
		\IF{$ N $ halts on $ x $}
		\RETURN{$ N'=N $}
		\ELSE
		\RETURN $ N' $ such that $ N' $ loops immediately.
		\ENDIF
	\end{algorithmic}
	
	Finally, let $ K $ accept $ N\#x $ if and only if $ M^* $ accepts $ N' $.\\
		
	\item \textit{Given a TM $ M $ and a string $ x $, does $ M $ ever read all of $ x $?}
	
	Not recursively enumerable - we can formulate the reduction $ \bar{MP} \leq L_3$. Given $ N\#x $, run $ N $ on $ x $. Create a TM $ N' $ with input $ y \in\Sigma^*\backslash\{\varepsilon\} $ with the properties dependent on whether or not $ N $ accepts $ x $:\\
	
	\begin{algorithmic}
		\IF{$ N $ accepts $ x $}
		\RETURN{$ N'\#y $ where $ N' $ rejects $ y $ without reading it}
		\ELSE
		\RETURN{$ N'\#y $ where $ N' $ reads all of $ y $ and then accepts}
		\ENDIF
	\end{algorithmic}
	Finally, let $ K $ accept $ N\#x $ if and only if $ M^* $ accepts $ N' $.\\
	
	\item \textit{Given two total TMs $ M_1 $ and $ M_2 $, is $ L(M1) = L(M2) $?}
	
	Not recursively enumerable - we can formulate the reduction $ \bar{MP} \leq L_3$. Given $ N\#x $, run $ N $ on $ x $. Create two TMs $ M_1,\: M_2$ with input $ y $ with the properties dependent on whether or not $ N $ accepts $ x $:\\
			
	\begin{algorithmic}
		\IF{$ N $ accepts $ x $}
		\RETURN{$ M_1$ accepts all $ y $, $ M_2 $ rejects all $ y $} ($ \implies L(M_1)=\Sigma^*\neq \varnothing=L(M_2) $)
		\ELSE
		\RETURN{$ M_1$ and $ M_2 $ both accept all $ y $} ($ \implies L(M_1)=\Sigma^*=L(M_2) $)
		\ENDIF
	\end{algorithmic}
	Finally, let $ K $ accept $ N\#x $ if and only if $ M^* $ accepts $ N' $.
	\end{enumerate}
