	\subsection*{Part  1: Strung Together}
		Let $ N = (Q, \Sigma, \Delta, S,F) $ be a NFA. Prove by structural induction that for any set of states $ A\subseteq Q $ and strings $ x,y\in\Sigma^* $ \[ \hat{\Delta}(\hat{\Delta}(A,x),y)=\hat{\Delta}(A,xy).\]
        \linesep
		\begin{proof}$ $\\
			\noindent \textbf{Base case:} $ y=\epsilon $
			\begin{align*}
			\hat{\Delta}(\hat{\Delta}(A,x),y)&=\hat{\Delta}(\hat{\Delta}(A,x),\epsilon)\\
			&=\hat{\Delta}(A,x)\bc{by definition of $\hat{\Delta}$}\\
			&=\hat{\Delta}(A,x\epsilon)\\
			&=\hat{\Delta}(A,xy)
			\end{align*}
			
			\noindent To simply notation, let $ B:=\hat{\Delta}(A,x)$.\\
			
			\noindent \textbf{Inductive hypothesis:} $y=w$ \[\hat{\Delta}(B,w)=\hat{\Delta}(\hat{\Delta}(A,x),w)=\hat{\Delta}(A,xw)\]
			
			\noindent \textbf{Inductive case:} $ y=wa $
			\begin{align*}
				\hat{\Delta}(\hat{\Delta}(A,x),wa)&=\hat{\Delta}(B,wa)\\
				&=\bigcup_{q\in\hat{\Delta}(B,w)}\Delta(q,a) &&\bc{by definition of $ \hat{\Delta} $}\\
				&=\bigcup_{q\in\hat{\Delta}(A,xw)}\Delta(q,a)&&\bc{by inductive hypothesis}\\
				&=\hat{\Delta}(A,xwa) &&\bc{by definition of $ \hat{\Delta} $}\\
				&=\hat{\Delta}(A,xy) &&\bc{substitute $y=wa$}
			\end{align*}
		\end{proof}
