	\subsection*{Part  1: Mboarbeb Cblaobsbuarbeb}
	Given a language $ A $ over the alphabet $ \Sigma $, we can define a new language
	\[Expand(A) = \{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in A \text{ and }b_i\in \Sigma\}.\]
	That is, $ Expand(A) $ is the set of strings generated by taking a string $ x $ from $ A $ and adding an extra, arbitrary character after every character in $ x $. For example, \[Expand(\{aa, bb\}) = \{aaaa, aaab, abaa, abab, baba, babb, bbba, bbbb\}.\]
	
	\noindent Using regular expressions, prove that the set of regular languages is closed under the Expand
	operation. In other words, given a regular expression $ \alpha $, show how to construct a regular expression $ \alpha' $ such that $ L(\alpha')=Expand(L(\alpha)) $, and prove that this is the case.
	
	\linesep
	\begin{proof}
		Given $ a\in \Sigma $, and $\alpha, \beta,\gamma $ are regular expressions, let $ G :\alpha\rightarrow \alpha'$ be a function defined as follows: 
		\begin{itemize}
			\item $ G(\varepsilon)=\varepsilon$, $ G(\varnothing)=\varnothing$
			\item $ G(a)=\sum_{b\in \Sigma}ab$
			\item $ G(\beta+\gamma)=G(\beta)+G(\gamma) $
			\item $ G(\beta\gamma) =G(\beta)G(\gamma)$
			\item $ G((\beta)^*) =(G(\beta))^*$
		\end{itemize}
		\bigskip
				
		We claim that $ L(G(\alpha))=Expand(L(\alpha)) $, and will prove it by structural induction.
		\begin{itemize}
			\item $ L(G(\varepsilon))=L(\varepsilon)=\{\varepsilon\}=Expand(\varepsilon)$	
			\item $ L(G(\varnothing))=L(\varnothing)=\varnothing=Expand(\varnothing)$
			\item If $ \alpha=a $, then 
			\begin{align*}
				L(G(\alpha))=L(G(a))&=L(\sum_{b\in \Sigma}ab)&&\bc{def of $ G $}\\
				&=\bigcup_{b\in\Sigma} L(ab)&&\bc{def of +}\\
				&=\{ab\mid b\in \Sigma\}&&\bc{def of $ \cup $}\\
				&=\{x=ab\mid b\in \Sigma, x=a\in \alpha\}&&\bc{def of $ \alpha $}\\
				&=Expand(\alpha)&&\bc{def of $ Expand(\cdot) $}
			\end{align*}
			\item If $ \alpha=\beta+\gamma$, then 
			\begin{align*}
				L(G(\beta+\gamma))&=L(G(\beta)+G(\gamma))&&\bc{def of $ G $}\\
				&=L(G(\beta))\cup L(G(\gamma))&&\bc{def of +}\\
				&=Expand(\beta)\cup Expand(\gamma)&&\bc{Inductive hypothesis}\\
				&= \{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \beta \text{ and }b_i\in \Sigma\}  \\
				&\phantom{==}\cup \{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \gamma \text{ and }b_i\in \Sigma\}&&\bc{def of $ Expand(\cdot) $}\\
				&=\{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \beta+\gamma \text{ and }b_i\in \Sigma\}&&\bc{def of $ \cup $}\\
				&=Expand(\beta+\gamma)&&\bc{def of $ Expand(\cdot) $}
			\end{align*}
			\item If $ \alpha=\beta+\gamma$, then 
			\begin{align*}
				L(G(\beta\gamma)) &=L(G(\beta)G(\gamma))&&\bc{def of $ G $}\\
				&=L(G(\beta))L(G(\gamma))&&\bc{def of L}\\
				&=Expand(\beta)Expand(\gamma)&&\bc{Inductive Hypothesis}\\
				&=\{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \beta \text{ and }b_i\in \Sigma\}\\
				&\phantom{==}\{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \gamma \text{ and }b_i\in \Sigma\}&&\bc{def of $ Expand(\cdot) $}\\
				&=\{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \beta\gamma \text{ and }b_i\in \Sigma\}&&\bc{set definition}\\
				&=Expand(\beta\gamma)&&\bc{def of $ Expand(\cdot) $}\\
			\end{align*}
			
			\item If $ \alpha=\beta+\gamma$, then 			
			\begin{align*}
				L(G((\beta)^*)) &=L((G(\beta))^*)&&\bc{def of $ G $}\\
				&=(L(G(\beta)))^*&&\bc{def of $ L $}\\
				&=(Expand(\beta))^*&&\bc{Inductive hypothesis}\\
				&=(\{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \beta \text{ and }b_i\in \Sigma\})^* &&\bc{def of $ Expand(\cdot) $}\\
				&=\{x = a_1b_1a_2b_2 \dots a_nb_n | y = a_1a_2 . . . a_n\in \beta^* \text{ and }b_i\in \Sigma\}\\
				&=Expand((\beta)^*)&&\bc{def of $ Expand(\cdot) $}
			\end{align*}
		\end{itemize}
	\end{proof}
